# doc/txt/url-sniper/create-relative-package/create-relative-package.spec.txt

	the specs for the create-relative-package functionality
	
	# Purpose:
	# ------------------------------------------------------
	creates a zip file with all the files specified in the 
	meta/.<<env_type>>.url-sniper with their relative file paths so that 
	the user could deploy the product by :
	mkdir -p <<target_product_instance_dir>>
	unzip -o <<zip_file>> -d <<target_product_instance_dir>>	

	# Prequisites
	# ------------------------------------------------------
	zip , perl , bash , xargs

	# Installation
	# ------------------------------------------------------
	install zip , perl , bash , xargs

	# Interdependancies
	# ------------------------------------------------------
	you might specify different file paths containing file 
	as follows: 
	bash src/bash/url-sniper/url-sniper.sh -a create-relative-package -i meta/.deploy

	# Usage:
	# ------------------------------------------------------
	bash src/bash/url-sniper/url-sniper.sh -a create-relative-package
